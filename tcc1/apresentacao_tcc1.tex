\section{Introdução}
% ---------------------------------------------------------------
\begin{frame}{Contexto}
  \begin{itemize}
    \item Apresente o cenário que originou o problema.
    \item Mostre dados/indicadores que comprovem a existência do problema.
    \item Indique quem é afetado e as consequências de não resolvê-lo.
  \end{itemize}
  \vfill
  {\footnotesize\itshape Dica: 1 slide, foco em situar a banca rapidamente.}
\end{frame}

\begin{frame}{Problema e Justificativa}
  \begin{block}{Problema}
    Enuncie em uma frase clara o problema que o trabalho ataca.
  \end{block}
  \begin{itemize}
    \item \textbf{Relevância acadêmica}: contribuição para a área.
    \item \textbf{Relevância prática}: benefícios para usuários/sociedade.
    \item \textbf{Lacuna}: o que ainda não foi resolvido.
  \end{itemize}
\end{frame}

\begin{frame}{Objetivos}
  \begin{block}{Objetivo principal}
    Frase única, com verbo no infinitivo (desenvolver, propor, avaliar\ldots).
  \end{block}
  \textbf{Objetivos secundários:}
  \begin{enumerate}
    \item Levantamento bibliográfico sobre o tema;
    \item Levantamento e análise de requisitos;
    \item Projeto da arquitetura da solução;
    \item Implementação de protótipo (no TCC2);
    \item Validação por testes/métricas/estudo de caso.
  \end{enumerate}
\end{frame}
